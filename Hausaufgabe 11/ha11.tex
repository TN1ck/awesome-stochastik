\section*{Aufgabe 1}
\begin{enumerate}
\item
$$\PP(\frac14 < X < \frac13) = F(\frac13) - F(\frac14) = \frac13 - \frac14 = \frac12$$
\begin{align*}
\PP(X < \frac12 \lor X > \frac23) &= \PP(X < \frac12) + \PP(\frac23 < X) \\
&= \PP(X < \frac12) + (\PP(X < 1) - \PP(X < \frac23) \\
&= F(\frac12) + (1 - F(\frac23)) \\
&= \frac12 + (1 - \frac23) = \frac12 + \frac13 = \frac56
\end{align*}

\end{enumerate}
